DROP TABLE IF EXISTS departments;
DROP TABLE IF EXISTS employees;

       id    serial primary key,
       name    text not null,
       age    integer,
       gender    text,
       department_id    integer
);
CREATE TABLE departments (
       id    serial primary key,
       name  text
);
